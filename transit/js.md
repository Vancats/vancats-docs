6. 手写 instanceof

```js
function myInstanceof(left, right) {
	const proto = right.prototype
	let curProto = Object.getPrototypeOf(left)
	while (curProto !== null) {
		if (curProto === proto)
			return true
		curProto = Object.getPrototypeOf(curProto)
	}
	return false
}
```
